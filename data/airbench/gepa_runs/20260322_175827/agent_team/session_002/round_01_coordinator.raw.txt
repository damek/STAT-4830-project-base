```python
#!/usr/bin/env python3
import argparse
import json
import os
import time
import torch
import torch.nn as nn
import torch.nn.functional as F
from torchvision import datasets, transforms

# --- setup_code ---
class BatchNorm(nn.BatchNorm2d):
    def __init__(self, num_features, eps=1e-05, momentum=0.1, affine=True, track_running_stats=True):
        super().__init__(num_features, eps=eps, momentum=momentum, affine=affine, track_running_stats=track_running_stats)

    def forward(self, x):
        return F.batch_norm(x, self.running_mean, self.running_var, self.weight, self.bias, self.training or not self.track_running_stats, self.momentum, self.eps)

# --- model_code ---
class ConvBlock(nn.Sequential):
    def __init__(self, in_channels, out_channels):
        super().__init__(
            nn.Conv2d(in_channels, out_channels, kernel_size=3, padding=1, bias=False),
            BatchNorm(out_channels),
            nn.ReLU(inplace=True),
            nn.MaxPool2d(2)
        )

class CIFAR10Model(nn.Module):
    def __init__(self, num_classes=10):
        super().__init__()
        self.features = nn.Sequential(
            ConvBlock(3, 64),
            ConvBlock(64, 128),
            ConvBlock(128, 256),
            ConvBlock(256, 512),
        )
        self.classifier = nn.Sequential(
            nn.Flatten(),
            nn.Linear(512 * 2 * 2, num_classes, bias=False)
        )

    def forward(self, x):
        return self.classifier(self.features(x))

# --- data_code ---
class CifarLoader:
    def __init__(self, data_dir, train=True, batch_size=128):
        transform = transforms.Compose([
            transforms.ToTensor(),
            transforms.Normalize((0.4914, 0.4822, 0.4465), (0.2023, 0.1994, 0.2010))
        ])
        dataset = datasets.CIFAR10(root=data_dir, train=train, download=True, transform=transform)
        self.loader = torch.utils.data.DataLoader(dataset, batch_size=batch_size, shuffle=train, num_workers=4, pin_memory=True)

    def __iter__(self):
        for x, y in self.loader:
            yield x.cuda(non_blocking=True), y.cuda(non_blocking=True)

# --- train_eval_code ---
def infer(model, loader, tta=False):
    model.eval()
    correct = 0
    total = 0
    with torch.no_grad():
        for x, y in loader:
            if tta:
                # Simple TTA: Original + Horizontal Flip
                out1 = model(x)
                out2 = model(torch.flip(x, [3]))
                out = (out1 + out2) / 2
            else:
                out = model(x)
            correct += (out.argmax(1) == y).sum().item()
            total += y.size(0)
    return correct / total

def train_one_epoch(model, loader, optimizer, criterion):
    model.train()
    for x, y in loader:
        optimizer.zero_grad(set_to_none=True)
        out = model(x)
        loss = criterion(out, y)
        loss.backward()
        optimizer.step()

def main():
    args = parse_args()
    
    # Preflight check
    if args.preflight:
        print("Preflight check passed.")
        return

    # Load data
    train_loader = CifarLoader(args.data_dir, train=True, batch_size=128)
    test_loader = CifarLoader(args.data_dir, train=False, batch_size=128)

    # Model
    model = CIFAR10Model().cuda()
    optimizer = torch.optim.SGD(model.parameters(), lr=0.1, momentum=0.9, weight_decay=5e-4)
    criterion = nn.CrossEntropyLoss()
    
    # Compilation
    model = torch.compile(model, mode="reduce-overhead")

    # Warmup
    for _ in range(args.warmup_trials):
        train_one_epoch(model, train_loader, optimizer, criterion)
        infer(model, test_loader)

    # Timed trials
    results = []
    for _ in range(args.trials):
        start = time.time()
        # Train for 1 epoch
        train_one_epoch(model, train_loader, optimizer, criterion)
        # TTA Eval
        acc = infer(model, test_loader, tta=True)
        end = time.time()
        results.append({"tta_val_accuracy": acc, "time_seconds": end - start})

    mean_acc = sum(r["tta_val_accuracy"] for r in results) / len(results)
    mean_time = sum(r["time_seconds"] for r in results) / len(results)

    output = {
        "mean_accuracy": mean_acc,
        "std_accuracy": 0.0,
        "mean_time_seconds": mean_time,
        "std_time_seconds": 0.0,
        "meets_target": mean_acc >= args.target_accuracy,
        "target_accuracy": args.target_accuracy,
        "trials": args.trials,
        "warmup_trials": args.warmup_trials,
        "per_trial": results,
        "torch_version": torch.__version__,
        "device_name": torch.cuda.get_device_name(0),
        "compiled": True
    }
    print(json.dumps(output))

# --- cli_code ---
def parse_args():
    parser = argparse.ArgumentParser()
    parser.add_argument("--data-dir", type=str, default="/vol/cifar10")
    parser.add_argument("--trials", type=int, default=1)
    parser.add_argument("--warmup-trials", type=int, default=1)
    parser.add_argument("--target-accuracy", type=float, default=0.94)
    parser.add_argument("--json-only", action="store_true")
    parser.add_argument("--preflight", action="store_true")
    return parser.parse_args()

if __name__ == "__main__":
    main()
```
